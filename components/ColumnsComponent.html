import { Component, OnInit, OnChanges, Input, EventEmitter, Output, ElementRef, Renderer2 } from '@angular/core';
// @ts-ignore
import ApexCharts from 'apexcharts';

@Component({
  selector: 'bobcat-columns',
  templateUrl: './columns.component.html',
  styleUrls: ['./columns.component.css']
})
export class ColumnsComponent implements OnInit, OnChanges {

  @Input('dataSeries') dataSeries:any;
  @Input('colors') colors:any;
  @Input('isMultipleSeries') isMultipleSeries:any;
  @Input('idComponent') idComponent:any;
  @Input('showYAxis') showYAxis:any;
  @Input('width') width:any;
  @Input('height') height:any;

  options:any = {
    chart: {
        height: 350,
        width: 300,
        type: 'bar',
        toolbar: {
          show: false
        },
        animations: {
          enabled: false
        }
    },
    plotOptions: {
        bar: {
            horizontal: false,
            barWidth: '100%',
            endingShape: 'rounded',
            distributed: false
        }
    },
    dataLabels: {
        enabled: false
    },
    stroke:{
        show: true,
        width: 2,
        colors: ['transparent']
    },
    series: [],
    xaxis: {
        categories: [],
        labels: {
            style: {
                colors: [],
                fontSize: '11px'
            }
        }
    },
    yaxis: {
        show:true,
        title: {
            text: ''
        }
    },
    fill: {
        opacity: 1

    },
    tooltip: {
        y: {
            formatter: function (val) {
                return val
            }
        }
    },
    colors: []
  }

  chart:any;

  constructor(
    private elRef: ElementRef,
    private renderer: Renderer2
  ) {

  }

  ngOnInit(){
    this.options.yaxis.show = this.showYAxis;
    this.options.chart.height = this.height;
    this.options.chart.width = this.width;

    //Render dinamico para el uso de multiples graficos de forma independiente.
    let div = this.renderer.createElement('div');
    div.id = this.idComponent;
    this.renderer.appendChild(this.elRef.nativeElement, div);

    //Set inicial de datos
    this.options.series = this.dataSeries.data;
    this.options.colors = this.colors;
    this.options.xaxis.categories = this.dataSeries.labels;
    this.options.plotOptions.bar.distributed = this.isMultipleSeries;

    //Creaci√≥n del graph dentro del div creado.
    this.chart = new ApexCharts(
        document.querySelector("#"+this.idComponent),
        this.options
    );
    this.chart.render();
  }
  ngOnChanges(){
    //Update de los datos
    if(this.chart){
      this.chart.updateSeries(this.dataSeries.data)
    }
  }

}
