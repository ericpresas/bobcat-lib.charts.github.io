import { Component, OnInit, OnChanges, Input, ElementRef, Renderer2 } from '@angular/core';
// @ts-ignore
import ApexCharts from 'apexcharts';

@Component({
  selector: 'bobcat-rt-line',
  templateUrl: './rt-line.component.html',
  styleUrls: ['./rt-line.component.css']
})
export class RtLineComponent implements OnInit, OnChanges {
  @Input('dataSeries') dataSeries:any;
  @Input('colors') colors:any;
  @Input('idComponent') idComponent:any;
  @Input('showYAxis') showYAxis:any;
  @Input('width') width:any;
  @Input('height') height:any;

  lastDate = 0;
  TICKINTERVAL = 86400000
  XAXISRANGE = 777600000

  options = {
      chart: {
          height: 350,
          width: 300,
          type: 'line',
          animations: {
              enabled: false,
              easing: 'linear',
              dynamicAnimation: {
                  speed: 1000
              }
          },
          toolbar: {
              show: false
          },
          zoom: {
              enabled: false
          }
      },
      dataLabels: {
          enabled: false
      },
      stroke: {
          curve: 'smooth'
      },
      series: [{
          data: []
      }],
      title: {
          text: '',
          align: 'left'
      },
      markers: {
          size: 0
      },
      xaxis: {
          type: 'datetime'
      },
      yaxis: {
          show:true
      },
      legend: {
          show: false
      },
      colors: []
  }

  chart:any;





  constructor(
    private elRef: ElementRef,
    private renderer: Renderer2
  ) { }

  ngOnInit() {
    this.options.yaxis.show = this.showYAxis;
    this.options.chart.height = this.height;
    this.options.chart.width = this.width;

    //Render dinamico para el uso de multiples graficos de forma independiente.
    let div = this.renderer.createElement('div');
    div.id = this.idComponent;
    this.renderer.appendChild(this.elRef.nativeElement, div);


    this.options.series = [
      {data: this.dataSeries}
    ];
    this.options.colors = this.colors;
    this.chart = new ApexCharts(
        document.querySelector("#"+this.idComponent),
        this.options
    );
    this.chart.render();
  }

  ngOnChanges() {
    if(this.chart){
      this.chart.updateSeries([{
              data: this.dataSeries
      }])
    }
  }

  private resetData(){
      // Alternatively, you can also reset the data at certain intervals to prevent creating a huge series
      this.dataSeries = this.dataSeries.slice(this.dataSeries.length - 10, this.dataSeries.length);
  }

}
